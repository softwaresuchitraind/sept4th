DROP SEQUENCE EMPLOYEE_SEQ;
create sequence employee_seq incremented by 1;

DROP TABLE  EMPLOYEE;
create table employee(id number, first_name varchar2(90),
			last_name varchar2(90),
			date_birth timestamp,
			joining_date date,
			username varchar2(90),
			password varchar2(90),
			gender_id number,
			latest_education_id number,
			project_id number,
constraint employee_pk primary key(id),
constraint employee_fk1 foreign key(GENDER_ID) REFERENCES GENDER(ID),
CONSTRAINT EMPLOYEE_FK2 FOREIGN KEY(LATEST_EDUCATION_ID) REFERENCES
					EDUCATION(ID),
CONSTRAINT EMPLOYEE_FK3 FOREIGN KEY(PROJECT_ID) REFERENCES
					PROJECT(ID),
			);
